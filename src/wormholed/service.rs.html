43async fn handle_cli_command(
44    ip: &IpP,
45    pods: &mut HashMap<String, Pod>,
46    command: Cli,
47    mut writer: CliTcpWriter,
48) {
49    let response_command = match command {
50        Cli::New(pod_args) => {
51            let mount_point_exists = pods.values().any(|p| p.get_mount_point() == &pod_args.path);
52
53            if mount_point_exists {
54                log::error!("This mount point already exist.");
55                Err(CliError::Message {
56                    reason: "This mount point already exist.".to_string(),
57                })
58            } else {
59                match commands::service::new(pod_args.clone()).await {
60                    Ok(pod) => {
61                        let name = pod.get_name().to_string();
62                        pods.insert(name.clone(), pod);
63                        Ok(CliSuccess::WithData {
64                            message: String::from("Pod created with success"),
65                            data: name,
66                        })
67                    }
68                    Err(e) => Err(e),
69                }
70            }
71        }
72        Cli::Start(pod_args) => commands::service::start(pod_args).await,
73        Cli::Stop(pod_args) => {
74            if let Some(pod) = pods.remove(&pod_args.name) {
75                commands::service::stop(pod).await
76            } else {
77                log::warn!("(TODO) Stopping a pod by path is not yet implemented");
78                Err(CliError::PodRemovalFailed {
79                    name: pod_args.name,
80                })
81            }
82        }
83        Cli::Remove(remove_arg) => {
84            let opt = if remove_arg.name != "." {
85                pods.remove(&remove_arg.name)
86            } else if remove_arg.path.inner != "." {
87                let key_to_remove = pods
88                    .iter()
89                    .find(|(_, pod)| pod.get_mount_point() == &remove_arg.path)
90                    .map(|(key, _)| key.clone());
91
92                key_to_remove.and_then(|key| pods.remove(&key))
93            } else {
94                log::error!("No pod name nor path were provided by RemovePod command");
95                None
96            };
97            if let Some(pod) = opt {
98                commands::service::remove(remove_arg, pod).await
99            } else {
100                Err(CliError::PodRemovalFailed {
101                    name: remove_arg.name,
102                })
103            }
104        }
105        Cli::Restore(mut resotre_args) => {
106            let opt_pod = if resotre_args.name == "." {
107                pods.iter()
108                    .find(|(_, pod)| pod.get_mount_point() == &resotre_args.path)
109            } else {
110                pods.iter().find(|(n, _)| n == &&resotre_args.name)
111            };
112            if let Some((_, pod)) = opt_pod {
113                resotre_args.path = pod.get_mount_point().clone();
114                commands::service::restore(
115                    pod.local_config.clone(),
116                    pod.global_config.clone(),
117                    resotre_args,
118                )
119            } else {
120                log::error!(
121                    "Pod at this path doesn't existe {:?}, {:?}",
122                    resotre_args.name,
123                    resotre_args.path
124                );
125                Err(CliError::PodRemovalFailed {
126                    name: resotre_args.name,
127                })
128            }
129        }
130        Cli::Apply(mut pod_conf) => {
131            // Find the good pod
132            let opt_pod = if pod_conf.name == "." {
133                pods.iter()
134                    .find(|(_, pod)| pod.get_mount_point() == &pod_conf.path)
135            } else {
136                pods.iter().find(|(n, _)| n == &&pod_conf.name)
137            };
138
139            //Apply new confi in the pod and check if the name change
140            let res = if let Some((name, pod)) = opt_pod {
141                pod_conf.path = pod.get_mount_point().clone();
142
143                match commands::service::apply(
144                    pod.local_config.clone(),
145                    pod.global_config.clone(),
146                    pod_conf.clone(),
147                ) {
148                    Err(err) => Err(err),
149                    Ok(_) => {
150                        match LocalConfig::read_lock(
151                            &pod.local_config.clone(),
152                            "handle_cli_command::apply",
153                        ) {
154                            Ok(local) => {
155                                if local.general.name != *name {
156                                    Ok(Some((local.general.name.clone(), name.clone())))
157                                } else {
158                                    Ok(None)
159                                }
160                            }
161                            Err(err) => Err(CliError::WhError { source: err }),
162                        }
163                    }
164                }
165            } else {
166                Err(CliError::Message {
167                    reason: format!(
168                        "This name or path doesn't existe in the hashmap: {:?}, {:?}",
169                        pod_conf.name, pod_conf.path
170                    ),
171                })
172            };
173
174            // Modify the name in the hashmap if it necessary
175            match res {
176                Ok(Some((new_name, old_name))) => {
177                    if let Some(pod) = pods.remove(&old_name) {
178                        pods.insert(new_name, pod);
179                        Ok(CliSuccess::Message("tt".to_owned()))
180                    } else {
181                        Err(CliError::Message {
182                            reason: "non".to_owned(),
183                        })
184                    }
185                }
186                Ok(None) => {
187                    todo!()
188                }
189                Err(err) => Err(err),
190            }
191        }
192        Cli::GetHosts(args) => {
193            if let Some(pod) = pods.get(&args.name) {
194                match pod.get_file_hosts(args.path) {
195                    Ok(hosts) => Ok(CliSuccess::WithData {
196                        message: "Hosts:".to_owned(),
197                        data: format!("{:?}", hosts),
198                    }),
199                    Err(error) => Err(CliError::PodInfoError { source: error }),
200                }
201            } else {
202                Err(CliError::PodNotFound)
203            }
204        }
205        Cli::Tree(args) => {
206            if let Some(pod) = pods.get(&args.name) {
207                match pod.get_file_tree_and_hosts(args.path) {
208                    Ok(tree) => Ok(CliSuccess::WithData {
209                        message: "File tree and hosts per file:".to_owned(),
210                        data: tree.to_string(),
211                    }),
212                    Err(error) => Err(CliError::PodInfoError { source: error }),
213                }
214            } else {
215                Err(CliError::PodNotFound)
216            }
217        }
218        Cli::Template(_template_arg) => todo!(),
219        Cli::Inspect => todo!(),
220        Cli::Status => Ok(CliSuccess::Message(format!("{}", ip.to_string()))),
221        Cli::Interrupt => todo!(),
222    };
223    let string_output =
224        response_command.map_or_else(|e| format!("CliError: {:?}", e), |a| a.to_string());
225    match writer.send(Message::Text(string_output)).await {
226        Ok(()) => log::debug!("Sent answer to cli"),
227        Err(err) => log::error!("Message can't send to cli: {}", err),
228    }
229}

231async fn get_cli_command(stream: tokio::net::TcpStream) -> WhResult<(Cli, CliTcpWriter)> {
232    // Accept the TCP stream as a WebSocket stream
233    let ws_stream = match accept_async(stream).await {
234        Ok(s) => s,
235        Err(e) => {
236            log::error!("get_cli_command: can't accept tcp stream: {}", e);
237            return Err(WhError::NetworkDied {
238                called_from: "get_cli_command::accept_tcp_stream".to_owned(),
239            });
240        }
241    };
242    let (writer, mut reader) = ws_stream.split();
243
244    // Read the first message from the stream
245    let message_data = match reader.next().await {
246        Some(Ok(msg)) => msg.into_data(),
247        Some(Err(e)) => {
248            log::error!("get_cli_command: invalid message: {}", e);
249            return Err(WhError::NetworkDied {
250                called_from: "get_cli_command".to_owned(),
251            });
252        }
253        None => {
254            log::error!("get_cli_command: can't get message from tcp stream");
255            return Err(WhError::NetworkDied {
256                called_from: "get_cli_command".to_owned(),
257            });
258        }
259    };
260
261    // Deserialize the message data into a Cli object
262    let cmd = match bincode::deserialize(&message_data) {
263        Ok(c) => c,
264        Err(e) => {
265            log::error!("get_cli_command: failed to deserialize message: {}", e);
266            return Err(WhError::NetworkDied {
267                called_from: "get_cli_command::deserialize_message".to_owned(),
268            });
269        }
270    };
271
272    Ok((cmd, writer))
273}

275const MAX_TRY_PORTS: u16 = 10;
276const MAX_PORT: u16 = 65535;
277
278custom_error::custom_error! {CliListenerError
279    ProvidedIpNotAvailable {ip: String, err: String} = "The specified address ({ip}) not available ({err})\nThe service is not starting.",
280    AboveMainPort {max_port: u16} = "Unable to start cli_listener (not testing ports above {max_port})",
281    AboveMaxTry {max_try_port: u16} = "Unable to start cli_listener (tested {max_try_port} ports)",
282}

284/// Listens for CLI calls and launch one tcp instance per cli command
285/// if `specific_ip` is not given, will try all ports starting from 8081 to 9999, incrementing until success
286/// if `specific_ip` is given, will try the given ip and fail on error.
287async fn start_cli_listener(
288    pods: &mut HashMap<String, Pod>,
289    specific_ip: Option<String>,
290    mut signals_rx: UnboundedReceiver<()>,
291) -> Result<(), CliListenerError> {
292    let mut ip: IpP = IpP::try_from(&specific_ip.clone().unwrap_or("127.0.0.1:8081".to_string()))
293        .expect("start_cli_listener: invalid ip provided");
294    println!("Starting CLI's Listener on {}", ip.to_string());
295
296    let mut port_tries_count = 0;
297    let mut listener = TcpListener::bind(&ip.to_string()).await;
298    while let Err(e) = listener {
299        if let Some(_) = specific_ip {
300            return Err(CliListenerError::ProvidedIpNotAvailable {
301                ip: ip.to_string(),
302                err: e.to_string(),
303            });
304        }
305        if ip.port >= MAX_PORT {
306            return Err(CliListenerError::AboveMainPort { max_port: MAX_PORT });
307        }
308        if port_tries_count > MAX_TRY_PORTS {
309            return Err(CliListenerError::AboveMaxTry {
310                max_try_port: MAX_TRY_PORTS,
311            });
312        }
313        log::warn!(
314            "Address {} not available due to {}, switching...",
315            ip.to_string(),
316            e
317        );
318        ip.set_port(ip.port + 1);
319        port_tries_count += 1;
320        log::debug!("Starting CLI's TcpListener on {}", ip.to_string());
321        listener = TcpListener::bind(&ip.to_string()).await;
322    }
323    log::info!("Started CLI's TcpListener on {}", ip.to_string());
324    let listener = listener.unwrap();
325
326    while let Some(Ok((stream, _))) = tokio::select! {
327        v = listener.accept() => Some(v),
328        _ = signals_rx.recv() => None,
329    } {
330        let (command, writer) = match get_cli_command(stream).await {
331            Ok(cmd) => cmd,
332            Err(e) => {
333                log::error!("cli_listener: error on getting command: {e}");
334                continue;
335            }
336        };
337        handle_cli_command(&ip, pods, command, writer).await;
338    }
339    Ok(())
340}

344#[tokio::main]
345async fn main() {
346    let (interrupt_tx, interrupt_rx) = mpsc::unbounded_channel::<()>();
347    let (signals_tx, signals_rx) = mpsc::unbounded_channel::<()>();
348
349    let mut pods: HashMap<String, Pod> = HashMap::new();
350
351    if env::args().any(|arg| arg == "-h" || arg == "--help") {
352        println!("Usage: wormholed <IP>\n\nIP is the node address, default at {DEFAULT_ADDRESS}");
353        return;
354    }
355
356    env_logger::init();
357
358    #[cfg(target_os = "windows")]
359    match winfsp_init() {
360        Ok(_token) => log::debug!("got fsp token!"),
361        Err(err) => {
362            log::error!("fsp error: {:?}", err);
363            std::process::exit(84)
364        }
365    }
366
367    let ip_string = env::args().nth(1);
368    let terminal_handle = tokio::spawn(terminal_watchdog(interrupt_tx));
369    let signals_task = tokio::spawn(handle_signals(signals_tx, interrupt_rx));
370    log::trace!("Starting service on {:?}", ip_string);
371    log::info!("Started");
372    let _ = start_cli_listener(&mut pods, ip_string, signals_rx).await;
373
374    terminal_handle.abort();
375    signals_task.await.unwrap();
376
377    log::info!("Stopping");
378    for (name, pod) in pods.into_iter() {
379        match pod.stop().await {
380            Ok(()) => log::info!("Stopped pod {name}"),
381            Err(e) => log::error!("Pod {name} can't be stopped: {e}"),
382        }
383    }
384    log::info!("Stopped");
385}

387// NOTE - old watchdog brought here for debug purposes
388pub async fn terminal_watchdog(tx: UnboundedSender<()>) {
389    let mut stdin = tokio::io::stdin();
390    let mut buf = vec![0; 1024];
391
392    while let Ok(read) = tokio::io::AsyncReadExt::read(&mut stdin, &mut buf).await {
393        // NOTE -  on ctrl-D -> quit
394        match read {
395            0 => {
396                let _ = tx.send(());
397                return;
398            }
399            _ => (),
400        };
401    }
402}

404async fn handle_signals(tx: UnboundedSender<()>, interrupt_rx: UnboundedReceiver<()>) {
405    #[cfg(unix)]
406    {
407        handle_signals_unix(tx, interrupt_rx).await;
408    }
409
410    #[cfg(windows)]
411    {
412        handle_signals_windows(tx, interrupt_rx).await;
413    }
414}
415
416#[cfg(unix)]
417async fn handle_signals_unix(tx: UnboundedSender<()>, mut interrupt_rx: UnboundedReceiver<()>) {
418    use tokio::signal::unix;
419
420    let mut sigint = unix::signal(unix::SignalKind::interrupt()).expect("failed to bind SIGINT");
421    let mut sigterm = unix::signal(unix::SignalKind::terminate()).expect("failed to bind SIGTERM");
422
423    log::info!("Unix signal handler initialised, waiting for SIGINT or SIGTERM…");
424
425    tokio::select! {
426        _ = sigint.recv() => {
427            log::info!("Quiting by Signal: SIGINT");
428            let _ = tx.send(());
429        }
430        _ = sigterm.recv() => {
431            log::info!("Quiting by Signal: SIGTERM");
432            let _ = tx.send(());
433        }
434        _ = interrupt_rx.recv() => {
435            log::info!("Quiting by Ctrl+D! (EOF)");
436            let _ = tx.send(());
437        }
438    }
439}

441#[cfg(windows)]
442async fn handle_signals_windows(tx: UnboundedSender<()>, mut interrupt_rx: UnboundedReceiver<()>) {
443    log::info!("Windows signal handler initialised…");
444
445    let mut sig_c = tokio::signal::windows::ctrl_c().expect("Failed to register ctrl_c");
446    let mut sig_break =
447        tokio::signal::windows::ctrl_break().expect("Failed to register ctrl_break");
448
449    tokio::select! {
450        _ = sig_c.recv() => {
451            log::info!("Quiting by Signal: CTRL+C");
452            let _ = tx.send(());
453        }
454        _ = sig_break.recv() => {
455            log::info!("Quiting by Signal: CTRL+BREAK");
456            let _ = tx.send(());
457        }
458        _ = interrupt_rx.recv() => {
459            log::info!("Quiting by Ctrl-Z (EOF)");
460            let _ = tx.send(());
461        }
462    }

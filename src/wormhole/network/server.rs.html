1use super::message::ToNetworkMessage;
2use crate::error::{CliError, CliResult};
3use std::{
4    collections::HashMap,
5    net::SocketAddr,
6    sync::{Arc, Mutex},
7};
8use tokio::{
9    net::{TcpListener, TcpSocket},
10    sync::mpsc::UnboundedReceiver,
11};
12pub type Tx = UnboundedReceiver<ToNetworkMessage>;
13pub type PeerMap = Arc<Mutex<HashMap<SocketAddr, Tx>>>;
14
15pub struct Server {
16    pub listener: TcpListener,
17    pub state: PeerMap,
18}
19
20impl Server {
21    pub async fn setup(addr: &str) -> CliResult<Server> {
22        let socket_addr: SocketAddr = addr.parse().map_err(|e| CliError::Server {
23            addr: addr.to_owned(),
24            err: std::io::Error::new(std::io::ErrorKind::InvalidInput, "Invalid ip address"),
25        })?;
26
27        let socket = TcpSocket::new_v4().map_err(|e| CliError::Server {
28            addr: addr.to_owned(),
29            err: e,
30        })?;
31        socket.set_reuseaddr(false).map_err(|e| CliError::Server {
32            addr: addr.to_owned(),
33            err: e,
34        })?;
35        socket.bind(socket_addr).map_err(|e| CliError::Server {
36            addr: addr.to_owned(),
37            err: e,
38        })?;
39        let listener = socket.listen(1024).map_err(|e| CliError::Server {
40            addr: addr.to_owned(),
41            err: e,
42        })?;
43
44        Ok(Server {
45            listener: listener,
46            state: PeerMap::new(Mutex::new(HashMap::new())),
47        })
48    }
49}

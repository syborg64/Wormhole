8#[derive(Debug, Parser, Serialize, Deserialize)] // requires `derive` feature
9#[command(name = "wormhole")]
10#[command(bin_name = "wormhole")]
11pub enum Cli {
12    /// Start the service
13    Start(StatusPodArgs),
14    /// Stop the service
15    Stop(StatusPodArgs),
16    /// Create a new network (template)
17    Template(TemplateArg),
18    /// Create a new pod and join a network if he have peers in arguments or create a new network
19    New(PodArgs),
20    /// Inspect a pod with its configuration, connections, etc
21    Inspect,
22    /// Get hosts for a specific file
23    GetHosts(GetHostsArgs),
24    /// Tree the folder structure from the given path and show hosts for each file
25    Tree(TreeArgs),
26    /// Remove a pod from its network
27    Remove(RemoveArgs),
28    /// Apply a new configuration to a pod
29    Apply(PodConf),
30    /// Restore many or a specifique file configuration  
31    Restore(PodConf),
32    /// Stops the service
33    Interrupt,
34}

36#[derive(Debug, clap::Args, Serialize, Deserialize, Clone)]

110#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, ValueEnum)]
111#[clap(rename_all = "lower")]
112pub enum Mode {
113    /// Simply remove the pod from the network without losing any data from the network
114    /// and leaving behind any data that was stored on the pod
115    Simple,
116    /// Remove the pod from the network without losing any data on the network,
117    /// and clone all data from the network into the folder where the pod was
118    /// making this folder into a real folder
119    Clone,
120    /// Remove the pod from the network and delete any data that was stored in the pod
121    Clean,
122    /// Remove this pod from the network without distributing its data to other nodes
123    Take,
124}

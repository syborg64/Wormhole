14pub async fn new(args: PodArgs) -> CliResult<Pod> {
15    let (global_config, local_config, server, mount_point) = pod_value(&args).await?;
16    Pod::new(
17        args.name.clone(),
18        global_config,
19        local_config.clone(),
20        mount_point,
21        server.clone(),
22        local_config.clone().general.address,
23    )
24    .await
25    .map_err(|e| CliError::PodCreationFailed { reason: e })
26}

28fn add_hosts(
29    mut global_config: GlobalConfig,
30    url: String,
31    mut additional_hosts: Vec<String>,
32) -> GlobalConfig {
33    if url.is_empty() && additional_hosts.is_empty() {
34        return global_config;
35    }
36
37    additional_hosts.push(url);
38    if global_config.general.peers.is_empty() {
39        global_config.general.peers = additional_hosts;
40    } else {
41        for host in additional_hosts {
42            if !global_config.general.peers.contains(&host) {
43                global_config.general.peers.push(host);
44            }
45        }
46    }
47    global_config
48}

50async fn pod_value(args: &PodArgs) -> CliResult<(GlobalConfig, LocalConfig, Arc<Server>, WhPath)> {
51    let local_path = args.path.clone().join(LOCAL_CONFIG_FNAME).inner;
52    let mut local_config: LocalConfig =
53        LocalConfig::read(&local_path).unwrap_or(default_local_config(&args.name));
54    if local_config.general.name != args.name {
55        //REVIEW - Change the name without notifying the user or return an error? I think it would be better to return an error
56        local_config.general.name = args.name.clone();
57    }
58    if local_config.general.address != args.ip {
59        local_config.general.address = args.ip.clone();
60    }
61    let server: Arc<Server> = Arc::new(Server::setup(&local_config.general.address).await?);
62
63    let global_path = args.path.clone().join(GLOBAL_CONFIG_FNAME).inner;
64    let global_config: GlobalConfig =
65        GlobalConfig::read(global_path).unwrap_or(default_global_config());
66    let global_config = add_hosts(
67        global_config,
68        args.url.clone().unwrap_or("".to_string()),
69        args.additional_hosts.clone().unwrap_or(vec![]),
70    );
71
72    Ok((global_config, local_config, server, args.path.clone()))
73}

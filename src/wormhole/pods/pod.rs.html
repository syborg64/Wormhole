67pub async fn initiate_connection(
68    peers_addrs: Vec<Address>,
69    server_address: Address,
70    tx: &UnboundedSender<FromNetworkMessage>,
71    rx: &mut UnboundedReceiver<FromNetworkMessage>,
72) -> Option<(FileSystemSerialized, Vec<Address>, PeerIPC, Vec<u8>)> {
73    if peers_addrs.len() >= 1 {
74        for first_contact in peers_addrs {
75            let first_ipc = PeerIPC::connect(first_contact.to_owned(), tx.clone()).await;
76
77            if let Some(ipc) = first_ipc {
78                if let Err(err) = ipc.sender.send((MessageContent::RequestFs, None)) {
79                    info!(
80                        "Connection with {first_contact} failed: {err}.\n
81                        Trying with next know address"
82                    );
83                    continue;
84                }
85
86                loop {
87                    info!("Awaiting response from {first_contact}");
88                    match rx.recv().await {
89                        // TODO: timeout here in case of peer malfunction
90                        Some(FromNetworkMessage {
91                            origin: _,
92                            content: MessageContent::FsAnswer(fs, mut peers_address, global_config),
93                        }) => {
94                            // remove itself from peers and first_connect because the connection is already existing
95                            peers_address.retain(|address| {
96                                *address != server_address && *address != first_contact
97                            });
98                            return Some((fs, peers_address, ipc, global_config));
99                        }
100                        Some(_) => {
101                            info!(
102                                "First message with {first_contact} failed: His answer is not the FileSystem, corrupted client.\n
103                                Trying with next know address"
104                            );
105                            break;
106                        }
107                        None => {
108                            info!("Empty response from {first_contact}...");
109                            continue;
110                        }
111                    };
112                }
113            }
114        }
115        info!("None of the known address answered correctly, starting a FS.")
116    }
117    None
118}

120fn register_to_others(peers: &Vec<PeerIPC>, self_address: &Address) -> std::io::Result<()> {
121    for peer in peers {
122        peer.sender
123            .send((MessageContent::Register(self_address.clone()), None))
124            .map_err(|err| std::io::Error::new(io::ErrorKind::NotConnected, err))?;
125    }
126    Ok(())
127}

138    pub async fn new(
139        name: String,
140        global_config: GlobalConfig,
141        local_config: LocalConfig,
142        mount_point: WhPath,
143        server: Arc<Server>,
144        server_address: Address,
145    ) -> io::Result<Self> {
146        let mut global_config = global_config;
147
148        log::info!("mount point {}", mount_point);
149        let (to_network_message_tx, to_network_message_rx) = mpsc::unbounded_channel();
150        let (from_network_message_tx, mut from_network_message_rx) = mpsc::unbounded_channel();
151        let (to_redundancy_tx, to_redundancy_rx) = mpsc::unbounded_channel();
152
153        global_config.general.peers.retain(|x| *x != server_address);
154
155        let mut peers = vec![];
156
157        let (arbo, next_inode, global_config_bytes) =
158            if let Some((fs_serialized, peers_addrs, ipc, global_config_bytes)) =
159                initiate_connection(
160                    global_config.general.peers.clone(),
161                    server_address.clone(),
162                    &from_network_message_tx,
163                    &mut from_network_message_rx,
164                )
165                .await
166            {
167                // TODO use global_config ?
168
169                peers = PeerIPC::peer_startup(peers_addrs, from_network_message_tx.clone()).await;
170                peers.push(ipc);
171                register_to_others(&peers, &server_address)?;
172
173                let mut arbo = Arbo::new();
174                arbo.overwrite_self(fs_serialized.fs_index);
175                let next_inode = arbo
176                    .iter()
177                    .fold(Arbo::first_ino(), |acc, (ino, _)| u64::max(acc, *ino))
178                    + 1;
179                (arbo, next_inode, Some(global_config_bytes))
180            } else {
181                let (arbo, next_inode) =
182                    generate_arbo(&mount_point, &server_address).expect("unable to index folder");
183                (arbo, next_inode, None)
184            };
185
186        let redundancy_target = global_config.redundancy.number;
187        let arbo: Arc<RwLock<Arbo>> = Arc::new(RwLock::new(arbo));
188        let local = Arc::new(RwLock::new(local_config));
189        let global = Arc::new(RwLock::new(global_config));
190
191        let network_interface = Arc::new(NetworkInterface::new(
192            arbo.clone(),
193            mount_point.clone(),
194            to_network_message_tx.clone(),
195            to_redundancy_tx.clone(),
196            next_inode,
197            Arc::new(RwLock::new(peers)),
198            local.clone(),
199            global.clone(),
200        ));
201
202        #[cfg(target_os = "linux")]
203        let disk_manager = Box::new(UnixDiskManager::new(&mount_point)?);
204        #[cfg(target_os = "windows")]
205        let disk_manager = Box::new(DummyDiskManager::new(&mount_point)?);
206
207        if let Some(global_config_bytes) = global_config_bytes {
208            if let Ok(perms) = Arbo::write_lock(&arbo, "Pod::new")?
209                .get_inode(GLOBAL_CONFIG_INO)
210                .map(|inode| inode.meta.perm)
211            {
212                let _ = disk_manager.new_file(&GLOBAL_CONFIG_FNAME.into(), perms);
213                disk_manager.write_file(&GLOBAL_CONFIG_FNAME.into(), &global_config_bytes, 0)?;
214            }
215        }
216        let fs_interface = Arc::new(FsInterface::new(
217            network_interface.clone(),
218            disk_manager,
219            arbo.clone(),
220        ));
221
222        // Start ability to recieve messages
223        let network_airport_handle = tokio::spawn(NetworkInterface::network_airport(
224            from_network_message_rx,
225            fs_interface.clone(),
226        ));
227
228        // Start ability to send messages
229        let peer_broadcast_handle = tokio::spawn(NetworkInterface::contact_peers(
230            network_interface.peers.clone(),
231            to_network_message_rx,
232        ));
233
234        let new_peer_handle = tokio::spawn(NetworkInterface::incoming_connections_watchdog(
235            server,
236            from_network_message_tx.clone(),
237            network_interface.peers.clone(),
238        ));
239
240        let peers = network_interface.peers.clone();
241
242        let redundancy_worker_handle = tokio::spawn(redundancy_worker(
243            to_redundancy_rx,
244            network_interface.clone(),
245            fs_interface.clone(),
246            redundancy_target,
247            server_address,
248        ));
249
250        Ok(Self {
251            name: name.clone(),
252            network_interface,
253            fs_interface: fs_interface.clone(),
254            mount_point: mount_point.clone(),
255            peers,
256            #[cfg(target_os = "linux")]
257            fuse_handle: mount_fuse(&mount_point, fs_interface.clone())?,
258            #[cfg(target_os = "windows")]
259            fsp_host: mount_fsp(&mount_point, fs_interface.clone())?,
260            network_airport_handle,
261            peer_broadcast_handle,
262            new_peer_handle,
263            local_config: local.clone(),
264            global_config: global.clone(),
265            redundancy_worker_handle,
266        })
267    }

271    pub fn get_file_hosts(&self, path: WhPath) -> Result<Vec<Address>, PodInfoError> {
272        let entry = Arbo::n_read_lock(&self.network_interface.arbo, "Pod::get_info")?
273            .get_inode_from_path(&path)
274            .map_err(|_| PodInfoError::FileNotFound)?
275            .entry
276            .clone();
277
278        match entry {
279            FsEntry::File(hosts) => Ok(hosts),
280            FsEntry::Directory(_) => Err(PodInfoError::WrongFileType {
281                detail: "Asked path is a directory (directories have no hosts)".to_owned(),
282            }),
283        }
284    }

287        let arbo = Arbo::n_read_lock(&self.network_interface.arbo, "Pod::get_info")?;

298    /// given ino is not checked -> must exist in arbo
299    fn recurse_tree(arbo: &Arbo, ino: InodeId, indentation: u8) -> Vec<TreeLine> {
300        let entry = &arbo
301            .n_get_inode(ino)
302            .expect("recurse_tree: ino not found")
303            .entry;
304        let path = arbo
305            .n_get_path_from_inode_id(ino)
306            .expect("recurse_tree: unable to get path");
307        match entry {
308            FsEntry::File(hosts) => vec![(indentation, ino, path, hosts.clone())],
309            FsEntry::Directory(children) => children
310                .iter()
311                .map(|c| Pod::recurse_tree(arbo, *c, indentation + 1))
312                .flatten()
313                .collect::<Vec<TreeLine>>(),
314        }
315    }

319    /// for a given file, will try to send it to one host, trying each until succes
320    async fn send_file_to_possible_hosts(
321        &self,
322        possible_hosts: &Vec<Address>,
323        ino: InodeId,
324    ) -> Result<(), PodStopError> {
325        let file_content =
326            self.fs_interface
327                .read_local_file(ino)
328                .map_err(|e| PodStopError::FileNotReadable {
329                    file: ino,
330                    reason: e.to_string(),
331                })?;
332        let file_content = Arc::new(file_content);
333
334        for host in possible_hosts {
335            let (status_tx, mut status_rx) = tokio::sync::mpsc::unbounded_channel::<WhResult<()>>();
336
337            self.network_interface
338                .to_network_message_tx
339                .send(ToNetworkMessage::SpecificMessage(
340                    (
341                        // NOTE - file_content clone is not efficient, but no way to avoid it for now
342                        MessageContent::RedundancyFile(ino, file_content.clone()),
343                        Some(status_tx.clone()),
344                    ),
345                    vec![host.clone()],
346                ))
347                .expect("to_network_message_tx closed.");
348
349            if let Some(Ok(())) = status_rx.recv().await {
350                self.network_interface
351                    .to_network_message_tx
352                    .send(ToNetworkMessage::BroadcastMessage(
353                        MessageContent::EditHosts(ino, vec![host.clone()]),
354                    ))
355                    .expect("to_network_message_tx closed.");
356                return Ok(());
357            }
358        }
359        Err(PodStopError::FileNotSent { file: ino })
360    }

362    /// Gets every file hosted by this pod only and sends them to other pods
363    async fn send_files_when_stopping(&self, arbo: &Arbo, peers: Vec<Address>) {
364        let address = if let Ok(local_conf_lock) = LocalConfig::read_lock(
365            &self.network_interface.local_config,
366            "send_files_when_stopping",
367        ) {
368            local_conf_lock.general.address.clone()
369        } else {
370            log::error!("send_files_when_stopping: can't lock local conf to get local address. No files sent.");
371            return;
372        };
373
374        futures_util::future::join_all(
375            arbo.files_hosted_only_by(&address)
376                .filter_map(|inode| {
377                    if inode.id == GLOBAL_CONFIG_INO
378                        || inode.id == LOCAL_CONFIG_INO
379                        || inode.id == ARBO_FILE_INO
380                    {
381                        None
382                    } else {
383                        Some(inode.id)
384                    }
385                })
386                .map(|id| self.send_file_to_possible_hosts(&peers, id)),
387        )
388        .await
389        .iter()
390        .for_each(|e| {
391            if let Err(e) = e {
392                log::warn!("{e:?}")
393            }
394        });
395    }

397    pub async fn stop(self) -> Result<(), PodStopError> {
398        // TODO
399        // in actual state, all operations (request from network other than just pulling the asked files)
400        // made after calling this function but before dropping the pod are undefined behavior.
401
402        // drop(self.fuse_handle); // FIXME - do something like block the filesystem
403
404        let arbo = Arbo::n_read_lock(&self.network_interface.arbo, "Pod::Pod::stop(1)")?;
405
406        let peers: Vec<Address> = self
407            .peers
408            .read()
409            .iter()
410            .map(|peer| peer.address.clone())
411            .collect();
412
413        self.send_files_when_stopping(&arbo, peers).await;
414        let arbo_bin = bincode::serialize(&*arbo).expect("can't serialize arbo to bincode");
415        drop(arbo);
416
417        self.network_interface
418            .to_network_message_tx
419            .send(ToNetworkMessage::BroadcastMessage(
420                MessageContent::Disconnect(
421                    LocalConfig::read_lock(&self.local_config, "pod::stop")?
422                        .general
423                        .address
424                        .clone(),
425                ),
426            ))
427            .expect("to_network_message_tx closed.");
428
429        let Self {
430            name: _,
431            network_interface: _,
432            fs_interface: _,
433            mount_point,
434            peers,
435            #[cfg(target_os = "linux")]
436            fuse_handle,
437            #[cfg(target_os = "windows")]
438            fsp_host,
439            network_airport_handle,
440            peer_broadcast_handle,
441            new_peer_handle,
442            redundancy_worker_handle: _,
443            global_config: _,
444            local_config: _,
445        } = self;
446
447        #[cfg(target_os = "linux")]
448        drop(fuse_handle);
449        #[cfg(target_os = "windows")]
450        drop(fsp_host);
451
452        fs::write(&mount_point.join(&ARBO_FILE_FNAME).inner, arbo_bin)
453            .map_err(|io| PodStopError::ArboSavingFailed { source: io })?;
454
455        *peers.write() = Vec::new(); // dropping PeerIPCs
456        network_airport_handle.abort();
457        new_peer_handle.abort();
458        peer_broadcast_handle.abort();
459        Ok(())
460    }
